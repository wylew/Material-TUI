08.4 — Snackbar / Toast
JavaScript 
// Add snackbar stack to your page:
// <div id="snackbar-stack"></div>

function toast(msg, type = 'ok', duration = 3500) {
  const prefixes = { ok:'[OK]', error:'[ER]', warning:'[WR]', info:'[II]' };
  const stack = document.getElementById('snackbar-stack');
  const el = document.createElement('div');
  el.className = `metui-snackbar metui-snackbar--${type}`;
  el.setAttribute('role', 'status');
  el.innerHTML = `
    <span class="metui-snackbar__prefix">${prefixes[type]}</span>
    <span class="metui-snackbar__msg">${msg}</span>
    <button class="metui-snackbar__dismiss"
            onclick="this.parentElement.remove()">[×]</button>
  `;
  stack.appendChild(el);
  setTimeout(() => el.remove(), duration);
}
